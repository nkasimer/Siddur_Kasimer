\chapter{תפילת שחרית}
\vspace{\baselineskip}
\firstword{בָּרוּךְ}
אַתָּה יְיָ אֱלֹהֵֽינוּ מֶֽלֶךְ הָעוֹלָם הַמַּעֲבִיר שֵׁנָה מֵעֵינָי וּתְנוּמָה מֵעַפְעַפָּי׃\\

\firstword{מוֹדֶה/מוֹדָה}
אֲנִי לְפָנֶיךָ מֶלֶךְ חַי וְקַיָּם \middot שֶׁהֶחֱזַרְתָּ בִּי נִשְׁמָתִי בְּחֶמְלָה \middot רַבָּה אֱמוּנָתֶךָ׃\\

\firstword{שְׁמַ֖ע יִשְׂרָאֵ֑ל יְיָ֥ אֱלֹהֵ֖ינוּ יְיָ֥ ׀ אֶחָֽד׃} \source{דברים ו}\\
בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד׃\\

\firstword{וְאָ֣הַבְתָּ֔}\source{דברים ו}
אֵ֖ת יְיָ֣ אֱלֹהֶ֑יךָ בְּכׇל־לְבָבְךָ֥ וּבְכׇל־נַפְשְׁךָ֖ וּבְכׇל־מְאֹדֶֽךָ׃\\

תּוֹרָ֥ה צִוָּה־לָ֖נוּ מֹשֶׁ֑ה מוֹרָשָׁ֖ה קְהִלַּ֥ת יַעֲקֹֽב׃\source{דברים לג}\\

שְׁמַ֣ע בְּ֭נִי מוּסַ֣ר אָבִ֑יךָ וְאַל־תִּ֝טֹּ֗שׁ תּוֹרַ֥ת אִמֶּֽךָ׃ \source{משלי א}
 וּמָצָא חֵן וְשֵֽׂכֶל טוֹב בְּעֵינֵי אֱלֹהִים וְאָדָם׃
 
 \firstword{אֱלֹהַי}
 נְצֹר לְשׁוֹנִי מֵרָע \middot וּשְׂפָתַי מִדַּבֵּר מִרְמָה׃\\
 אַל תְּבִיאֵנוּ לֹא לִידֵי חֵטְא \middot וְלֹא לִידֵי נִסָּיוֹן׃\\
 
 הַרְחִיקֵנִי מֵאָדָם רָע וּמֵחָבֵר רָע וְתִגְמְלֵנִי חֲסָדִים טוֹבִים׃\\
 
 \firstword{בָּרוּךְ}
 אַתָּה יְיָ אֱלֹהֵֽינוּ מֶֽלֶךְ הָעוֹלָם שֶׁעָשַֽׂנִי יִשְׂרָאֵל׃\\

\firstword{וְאָ֣הַבְתָּ֔}\source{דברים ו}

\firstword{בָּרוּךְ}
אַתָּה יְיָ אֱלֹהֵֽינוּ מֶֽלֶךְ הָעוֹלָם הַמַּפִּיל חֶבְלֵי שֵׁנָה עַל עֵינַי וּתְנוּמָה עַל עַפְעַפָּי׃ וִיהִי רָצוֹן מִלְּפָנֶֽיךָ יְיָ אֱלֹהַי וֵאלֹהֵי אֲבוֹתַי שֶׁתַּשְׁכִּיבֵֽנִי לְשָׁלוֹם וְתַעֲמִידֵֽנִי לְשָׁלוֹם׃\\

בָּרוּךְ יְיָ בַּיּוֹם \middot בָּרוּךְ יְיָ בַּלָּֽיְלָה׃ בָּרוּךְ יְיָ בְּשׇׁכְבֵֽנוּ \middot בָּרוּךְ יְיָ בְּקוּמֵֽנוּ׃\\

בְּיָדְךָ֮ \source{תהלים לא}אַפְקִ֢יד ר֫וּחִ֥י פָּדִ֖יתָ אוֹתִ֥י יְיָ֗ אֵ֣ל אֱמֶֽת׃\\

לִישׁוּעָֽתְךָ֖ קִוִּ֥יתִי יְיָֽ׃\source{בראשית מט}
